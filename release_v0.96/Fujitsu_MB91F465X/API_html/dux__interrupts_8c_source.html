00083  * Interrupt routine to read out the current ADC values from the micro controller slider on the 
00084  * Fujitsu board. Sends over FlexRay the value to the other node. Hardware dependant code!
00085  */
00086 __interrupt void ADC_int(void){     
00087          uint16 ad_data9;
00088          uint8 led;
00089          uint8 ad_ch;
00090 
00091          DUXCPUWaitOneNop;
00092          DUXCPUWaitOneNop;
00093          DUXCPUWaitOneNop;
00094          DUXCPUWaitOneNop;
00095          DUXCPUWaitOneNop;
00096 
00097          DUXWatchDogReset; 
00098          ad_ch = IO_AD.IO_ADCS0.byte & 0x1f;
00099          
00100          if(ad_ch==0x9){ /* the left one increases the local leds*/ 
00101                   DUXCPUWaitOneNop;
00102                   DUXCPUWaitOneNop;
00103                   ad_data9 = 0x0000;
00104                   ad_data9 = IO_AD.IO_ADCR01.hword;
00105                   led = (uint8) ad_data9;
00106                   IO_PDR14.byte = led;
00107                   DUXWatchDogReset;          
00108          }
00109          else if(ad_ch==0xa){ /* increase the other leds over flexray*/
00110                   DUXCPUWaitOneNop;
00111                   DUXCPUWaitOneNop;
00112 #ifdef NODE_1              
00113                   DUXWatchDogReset;
00114                   _sendFrameBuffer[0] = IO_AD.IO_ADCR01.hword; /* transfer a mix of adjustable byte and a string set for endianess check */               
00115                   
00116                   DUXWatchDogReset;          
00117                   DUXFXSendData(1, 2, _sendFrameBuffer, sizeof(_sendFrameBuffer));        /* take care not to send more than is allowed in size */
00118 #endif            
00119          }
00120          else{
00121                   DUXWatchDogReset;
00122          }
00123          
00124          IO_AD.IO_ADCS1.bit.INT = 0;                  /* Int frag clear */
00125          DUXWatchDogReset;
00126          DUXCPUWaitOneNop;
00127          IO_AD.IO_ADCS1.bit.STRT = 0x1;      /* set A/D control start */
00128 
00129 }
