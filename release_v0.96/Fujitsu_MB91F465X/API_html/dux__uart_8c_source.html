00041 /* DUXUartInit(void)
00042  * Implements the initialization of the UART.
00043  */
00044 void DUXUartInit(void)
00045 {
00046          DUXWatchDogReset; 
00047          BGR04 = 2499;      /*  9600 Baud @ 24MHz */
00048     /* BGR04 = 1249; */  /* 19200 Baud @ 24MHz */
00049     /* BGR04 = 624;  */  /* 38400 Baud @ 24MHz */     
00050          SCR04 = 0x17; /* 8 bit, clear reception errors, Tx & Rx enabled */
00051     SMR04 = 0x0D; /* Mode 0, Reset Counter, Reset UART, SOT0 enabled */
00052     SSR04 = 0x00; /* LSB first, TDRE bit read only default 1 and written 0       */
00053     
00054          PORT2_PFR19 = (PORT2_PFR19 & 0xFC) | 0x03; /* enable UART: SIN,SOT for async. transfer */
00055          PORT2_EPFR19 = 0x00; /* enable UART */
00056          
00057          DUXWatchDogReset; 
00058 }

00115 void _DUXPutCharacter(char ch) /* sends a char */
00116 {
00117   while (SSR04_TDRE == 0) DUXWatchDogReset; /* wait for transmit buffer empty    */
