00102 /* DUXFXStartup()
00103  * This function executes FlexRay synchronization process. 
00104  * Startup process and transition to Normal Active state.
00105  * Node performs as Startup node.
00106  */
00107 void DUXFXStartup(void)
00108 {
00109          DUXWatchDogReset; 
00110 
00111          /*** For Start Up Node only ***/
00112 
00113          /* WAKEUP command */
00114           FxR_SUC1_SUCC1 = DUX_SUC_CMD_WAKEUP;
00115          /* Switch into config state !*/              
00116                   
00117          /* Check the POC busy state with SUCC1:PBSY[7] */
00118          while((FxR_SUC1_SUCC1 & 0x00000080) == 0x00000080)    DUXWatchDogReset;
00119 
00120          /* Wait for READY state, after WAKEUP completion */
00121          while( (FxR_SUC2_CCSV&0x0000003F) != DUX_CCSV_POCS_READY /*0x01*/ ) DUXWatchDogReset; 
00122 
00123          /* COLDSTART command */
00124           FxR_SUC1_SUCC1 = DUX_SUC_CMD_ALLOW_COLDSTART;
00125 
00126          /* Check the POC busy state with SUCC1:PBSY[7] */
00127          while((FxR_SUC1_SUCC1 & 0x00000080) == 0x00000080)    DUXWatchDogReset;
00128 
00129          /* RUN command (start up processing) */
00130          FxR_SUC1_SUCC1 = DUX_SUC_CMD_RUN;
00131 
00132          /* Wait for NORMALACTIVE state, after RUN command execution  */
00133          while( (FxR_SUC2_CCSV&0x0000003F) != DUX_CCSV_POCS_NML_ACT /*0x02*/ )
00134          {
00135                   DUXCPUWaitOneNop;
00136                   /* Reset the watchdog timer */
00137                   DUXWatchDogReset; 
00138          }
00139 
00140          /* Finish the Startup */
00141          return;
00142 }

00144 /* DUXFXSendData(uint16 slotID, uint8 channel, const uint8* data, uint8 payload)
00145  * Given a slot number, the channel, the data and the payload 
00146  * in byte (!) the frame is transmitted on the bus. If no 
00147  * buffer can be found for sending the routine returns FALSE.
00148  */
00149 boolean DUXFXSendData(uint16 slotID, uint8 channel, const uint8* data, uint8 payload){    
00150          sint16 myBufferIndex;
00151          uint8 len, i;     
00152          uint32 marshallData = 0;
00153 
00154          DUXWatchDogReset; 
00155          
00156          myBufferIndex = -1;
00157          /* look if slot ID matches to a static buffer */      
00158          myBufferIndex = _DUXFXfindBufferForStaticSenderSlot(slotID,channel);
00159          if ((myBufferIndex<0)&&(channel == DUX_CHANNEL_AB)) return FALSE; /* there is no buffer reserved for this slot !*/
00160 
00161          DUXWatchDogReset; 
00162          
00163          /* Now look up the dynamic segment */        
00164          if (myBufferIndex<0){
00165                   myBufferIndex = _DUXFXfindBufferForDynamicSenderSlot(slotID,channel);            
00166          }
00167          
00168          DUXWatchDogReset; 
00169          
00170          if (myBufferIndex<0) return FALSE; /* there is no buffer reserved for this slot !*/
00171          
00172          /* marshall data */
00173          len = payload/4;
00174          for(i=0; i < len; i++) { 
00175                   DUXWatchDogReset;                   
00176                   marshallData = 0;
00177                   marshallData = data[(i*4)+3] << 24 | data[(i*4)+2] << 16 | data[(i*4)+1] << 8 | data[(i*4)+0];
00178                   _DUXFXTransmitData(i, marshallData);                  
00179          }         
00180          /* stuff in the rest */
00181          if (payload %4 != 0){
00182                   for(i=0; i < payload%4; i++) {
00183                            DUXWatchDogReset;                            
00184                            marshallData = 0;                   
00185                            marshallData |= data[(len*4) + i] << (8*i);
00186                            DUXWatchDogReset;
00187                   }
00188                   _DUXFXTransmitData((len*4), marshallData);
00189          }
00190 
00191          DUXWatchDogReset;          
00192          _DUXFXTransmitControl(myBufferIndex);        /* activate sending of buffer */
00193          
00194          return TRUE;
00195 }

00273  * Private method implementation for cc initialization of FlexRay controller registers.
00274  * Initialization must be executed in CC State "CONFIG" or "DEFALUT CONFIG"
00275  */
00276 void _DUXFXConfigInit(void)
00277 {
00278          uint32 i = 0; /* Loop Counter "i" */
00279 
00280          /* FlexRay PLL  : FlexRay system Clock 80MHz */        
00281          PLL2_DIVM = 0x01;
00282          PLL2_DIVN = 0x13;
00283          PLL2_DIVG = 0x00;
00284          PLL2_MULG = 0x00;
00285 
00286          PLL2_CLKR2 |= 0x04;
00287          PLL2_CLKR2 |= 0x02;
00288 
00289          while(1)
00290          {
00291                   /*  Wait for Ocsillation stability (wait more than 600us) */
00292                   /* 19200 is for the 10MHz Oscillator */
00293                   if(i > 19200)
00294                   {
00295                            break;
00296                   }
00297                   i++;
00298                   DUXWatchDogReset;                   
00299          } 
00300 
00301          FxR_PRT_PRTC1     = DUX_FLX_INIT_PRTC1;      /* PRT1 register */
00302          FxR_PRT_PRTC2     = DUX_FLX_INIT_PRTC2;      /* PRT2 register*/
00303          FxR_MHDC          = DUX_FLX_INIT_MHDC;       /* MHD  register*/
00304          FxR_GTU1_GTUC1  = DUX_FLX_INIT_GTUC1;        /* GTU1 register*/
00305          FxR_GTU1_GTUC2  = DUX_FLX_INIT_GTUC2;        /* GTU2 register*/
00306          FxR_GTU1_GTUC3  = DUX_FLX_INIT_GTUC3;        /* GTU3 register*/
00307          FxR_GTU1_GTUC4  = DUX_FLX_INIT_GTUC4;        /* GTU4 register*/
00308          FxR_GTU1_GTUC5  = DUX_FLX_INIT_GTUC5;        /* GTU5 register*/
00309          FxR_GTU1_GTUC6  = DUX_FLX_INIT_GTUC6;        /* GTU6 register*/
00310          FxR_GTU1_GTUC7  = DUX_FLX_INIT_GTUC7;        /* GTU7 register*/
00311          FxR_GTU1_GTUC8  = DUX_FLX_INIT_GTUC8;        /* GTU8 register*/
00312          FxR_GTU1_GTUC9  = DUX_FLX_INIT_GTUC9;        /* GTU9 register*/
00313          FxR_GTU1_GTUC10 = DUX_FLX_INIT_GTUC10;  /* GTU10 register*/

00318

00323             
00324          while((FxR_SUC1_SUCC1 & 0x00000080) == 0x00000080)    DUXWatchDogReset;

00328 /* _DUXFXCreateMessageRam()
00329  * Private method implementation for the setup
00330  * of message buffers for transmit and receive.
00331  */
00332 void _DUXFXCreateMessageRam(void)
00333 {   
00334          uint32 index;
00335          
00336          /* Configurate Message RAM partation: MRC */
00337          /* Security                                                                        MRC:SEC[25:24]
00338             Last Message Buffer                                           MRC:LCB[23:16]
00339             The first Buffer Number of FIFO             MRC:FFB[15:8]
00340             The first Buffer Number for Dynamic Slot MRC:FDB[0:7]
00341          */
00342          FxR_MHD_MRC = DUX_FLX_INIT_MRC; 
00343 
00344          /**** Set Message RAM Header ***/
00345          DUXWatchDogReset;          
00346          for( index=0; index<DUX_MSG_BUFF_NUM; index++)
00347          {
00348                   DUXWatchDogReset;                   
00349                   /* Write Header Section Register 1-3: WRHS1-3 setting */
00350                   if(typ_FLX_msgobj[index].msgval == DUX_FLX_MSGOBJ_USE){
00351                            FxR_IBF_WRHS1 = typ_FLX_msgobj[index].WRHS1;
00352                            FxR_IBF_WRHS2 = typ_FLX_msgobj[index].WRHS2;
00353                            FxR_IBF_WRHS3 = typ_FLX_msgobj[index].WRHS3;
00354 
00355                            /* Header Section Load */
00356                            /* Wait for the release of Input Host Buffer */
00357                            while ((FxR_IBF_IBCR & 0x00008000) != 0) DUXWatchDogReset;
00358                   
00359                            /* Header Section Load request LHSH[0] */
00360                            FxR_IBF_IBCM = 0x00000001;
00361 
00362                            /* Request for the transmission to MB IBRH[6:0] */
00363                            FxR_IBF_IBCR = ( index & 0x7F );
00364                   }
00365          }
00366                   
00367          return;
00368 }

00451 /* _DUXFXConfigFix()
00452  * Next process of FlexRay CC & Message buffer.
00453  * State transition to READY and Configuration Lock.
00454  * This function must be executed after configuration.
00455  */
00456 void _DUXFXConfigFix(void)
00457 {
00458          DUXWatchDogReset; 
00459          
00460          /* The oder is Commnad request is CONFIG->READY->WAKEUP->COLDSTART
00461             To Transit POC state to NORMAL_ACTIVEm, then Communication is established */
00462          
00463          /* wait for READY state */
00464          while( (FxR_SUC2_CCSV&0x0000003F) != DUX_CCSV_POCS_READY /*0x01*/ ) 
00465          {
00466                   DUXWatchDogReset;                   
00467                   
00468                   /* Check the POC busy state with SUCC1:PBSY[7] */
00469                   while((FxR_SUC1_SUCC1 & 0x00000080) == 0x00000080)    DUXWatchDogReset;

00473                   FxR_GIF1_LCK = 0x000000CE; /* 1st code */
00474                   FxR_GIF1_LCK = 0x00000031; /* 2nd code */
00475                   
00476                   /* Transit POC state to READY (from CONFIG) */
00477                   FxR_SUC1_SUCC1 = DUX_FLX_READY_SUCC1;
00478 
00479                   DUXWatchDogReset;

00486 /* _DUXFXTransferReceivedBufferToRam()
00487  * Implements the transfer of the shadow buffers to the memory.
00488  */
00489 void _DUXFXTransferReceivedBufferToRam(uint16 buf_no)
00490 {
00491          DUXWatchDogReset;          
00492          
00493          /* 1 Request header and data transmitting  Message RAM -> Shadow Buffer 
00494                    (REQ Command) */ 
00495          /* Set transmitting Mask */
00496          /* Request Shadow Data RDSS[1]A Request Shadow Header RHSS[0] */
00497          FxR_OBF_OBCM |= (0x00000002|0x00000001);
00498 
00499          /* Transmitting to Shadow Buffer */
00500          /* The number of Message buffer as source OBRS[6:0] */
00501          FxR_OBF_OBCR = (buf_no & 0x007F);
00502          /* Request for messgae transmitting REQ[9] */
00503          FxR_OBF_OBCR |= (0x00000200);
00504 
00505          DUXWatchDogReset; 
00506          /* check the OBF (Shadow) busy */
00507          /* Busy state  OBSYS[15] */
00508          while( (FxR_OBF_OBCR & 0x00008000) != 0 )    DUXWatchDogReset;
00509 
00510          /* 2 exchange shadow buffer and host buffer to access receive data from CPU 
00511                    (VIEW command) */
00512 
00513          /* Request for message view [8] */
00514          FxR_OBF_OBCR |= (0x00000100);
00515 
00516          DUXWatchDogReset; 
00517 }
